  async getHeroList() {
    const hero_list = await window.baseRequest.getHeroList();
    const position = await window.baseRequest.getHeroListPosition();
    const hero_rank = await window.baseRequest.getHeroRank();

    hero_list.hero = _.map(hero_list.hero,(hero)=>{
      // ﾕｾﾎｻ魹ｽﾏｲ魹ｽ
      hero.position = _.keys(_.get(position,'list.'+hero.heroId));
      hero.position_info = [];
      _.forEach(hero.position,(pos)=>{
        hero.position_info.push(_.find(window.BaseDataManager.hero_pos,{type:pos}))
      });

      // ﾖｰﾒｵ魹ｽﾏｲ魹ｽ
      hero.roles_info = [];
      _.forEach(hero.roles,(pos)=>{
        hero.roles_info.push(_.find(window.BaseDataManager.hero_job,{type:pos}))
      });

      hero.record_data = _.get(hero_rank,'list.'+hero.heroId);

      return hero
    });
    return hero_list.hero;

    let logoSizeParam = '/0';
    if (typeof (o) === 'string') {
      if (!this.judgeEndStr(o, logoSizeParam)) {
        if (o.indexOf('qtl_user') !== -1 || o.indexOf('//p.qpic.cn/qtlinfo') !== -1) {

      for (let i = 0, j = o.length; i < j; ++i) {
        let obj = o[i];
        let logoUrl = obj.logo_url;
        if (!window.this.judgeEndStr(logoUrl, logoSizeParam)) {
          if (logoUrl.indexOf('qtl_user') !== -1 || logoUrl.indexOf('//p.qpic.cn/qtlinfo') !== -1) {
